sev_dif <- mb |>

mb2 <- mb |> 
  filter(!is.na(prod), !is.na(sev)) |>

Preparation of the data

correlation_data <- mb2 %>%
  group_by(cod) %>%
  summarise(r_sev_yld = cor(sev, prod),
            n = n(),
            Z_sev_yld = 0.5*log((1+r_sev_yld)/(1-r_sev_yld)),
            V = 1/(n-3),
            vd_sev_yld = 1/(2*(1-r_sev_yld)/n-3)) |> 
  filter(r_sev_yld != "NA") |> 
  filter(n > 4)
